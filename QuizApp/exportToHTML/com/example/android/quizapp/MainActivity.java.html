27       //This method first checks that all the questions have been answered then assigns a score. 
28       public void gradeQuiz(View view) { 
29           grade = 0; 
30           //Validating Question 1 
31           questionOne = findViewById(R.id.questionOneText); 
32           String questionOneInput = questionOne.getText().toString(); 
33           if (questionOneInput.isEmpty()) { 
34               Toast.makeText(this, "Please answer all the questions first!", Toast.LENGTH_SHORT).show(); 
35               return; 
36           } 
37           //Validating Question 2 
38           radioGroupQ2 = findViewById(R.id.radioGroupQ2); 
39           if (radioGroupQ2.getCheckedRadioButtonId() == -1) { 
40               Toast.makeText(this, "Please answer all the questions first!", Toast.LENGTH_SHORT).show(); 
41               return; 
42           } 
43           //Validating Question 3 
44           radioGroupQ3 = findViewById(R.id.radioGroupQ3); 
45           if (radioGroupQ3.getCheckedRadioButtonId() == -1) { 
46               Toast.makeText(this, "Please answer all the questions first!", Toast.LENGTH_SHORT).show(); 
47               return; 
48           } 
49           //Validating Question 4 
50           questionFour = findViewById(R.id.questionFourText); 
51           String questionFourInput = questionFour.getText().toString(); 
52           if (questionFourInput.isEmpty()) { 
53               Toast.makeText(this, "Please answer all the questions first!", Toast.LENGTH_SHORT).show(); 
54               return; 
55           } 
56           //Validating Question 5 
57           q5OptionA = findViewById(R.id.question5OptionA); 
58           q5OptionB = findViewById(R.id.question5OptionB); 
59           q5OptionC = findViewById(R.id.question5OptionC); 
60           q5OptionD = findViewById(R.id.question5OptionD); 
61           if (!(q5OptionA.isChecked() || q5OptionB.isChecked() || q5OptionC.isChecked() || q5OptionD.isChecked())) { 
62               Toast.makeText(this, "Please answer all the questions first!", Toast.LENGTH_SHORT).show(); 
63               return; 
64           } 
65           //Validating Question 6 
66           q6OptionA = findViewById(R.id.q6OptionA); 
67           q6OptionB = findViewById(R.id.q6OptionB); 
68           q6OptionC = findViewById(R.id.q6OptionC); 
69           q6OptionD = findViewById(R.id.q6OptionD); 
70           if (!(q6OptionA.isChecked() || q6OptionB.isChecked() || q6OptionC.isChecked() || q6OptionD.isChecked())) { 
71               Toast.makeText(this, "Please answer all the questions first!", Toast.LENGTH_SHORT).show(); 
72               return; 
73           } 
74           //Once we have verified that all the questions have been answered we now mark the quiz 
75           // Mark Question 1 
76           if (questionOneInput.equalsIgnoreCase("incorrectly")) { 
77               grade = grade + 1; 
78           } 
79           //Mark Question 2 
80           radioButtonQ2 = findViewById(R.id.radio_optionD); 
81           if (radioButtonQ2.isChecked()) { 
82               grade = grade + 1; 
83           } 
84           //Mark Question 3 
85           radioButtonQ3 = findViewById(R.id.q3Radio_optionC); 
86           if (radioButtonQ3.isChecked()) { 
87               grade = grade + 1; 
88           } 
89           //Mark Question 4 
90           if (questionFourInput.equalsIgnoreCase("alphabet")) { 
91               grade = grade + 1; 
92           } 
93           //Mark Question 5. For this one gets a 0 if one option is wrong and 3 points if all options are right. 
94           if ((q5OptionB.isChecked() && q5OptionD.isChecked()) && !q5OptionA.isChecked() && !q5OptionC.isChecked()) { 
95               grade = grade + 3; 
96           } 
97           if (q6OptionA.isChecked() && q6OptionC.isChecked() && q6OptionD.isChecked() && !q6OptionB.isChecked()) { 
98    
99               grade = grade + 3; 
100          } 
101          grade = grade * 10; 
102          if (grade == 100) { 
103              Toast.makeText(this, "Congratulations!!!\nYou have a perfect SCORE!", Toast.LENGTH_LONG).show(); 
104          } else { 
105              Toast.makeText(this, "Well done, you scored, " + grade + "%", Toast.LENGTH_LONG).show(); 
106          } 
107      }
